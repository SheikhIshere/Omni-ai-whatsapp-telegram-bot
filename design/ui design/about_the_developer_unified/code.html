@app.post("/v1/agent/execute")
async def process_workflow(payload: AgentPayload):
    # 1. Start Trace
    with tracer.start_span("agent_thought"):
        context = await vector_store.search(
            payload.query
        )
        
    # 2. LLM Reasoning
    response = await openai.chat.completions.create(
        model="gpt-4-turbo",
        messages=build_prompt(context, payload)
    )
    
    # 3. Handle Tool Calls
    if response.choices[0].message.tool_calls:
        return await handle_tools(response)

    return {"status": "success"}
